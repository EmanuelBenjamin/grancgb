CREATE TABLE CGB.SEG_EMPRESAS (
  ID_EMP INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
  NOMBRE_EMP VARCHAR(30) NOT NULL,
  PAIS_EMP VARCHAR(3) NOT NULL,
  EMPRESA_EMP VARCHAR(3) NOT NULL,
  DEPARTAMENTO_EMP INT(11) UNSIGNED NOT NULL,
  PERIODO_EMP VARCHAR(5) NOT NULL,
  TIPO_CONTA_EMP VARCHAR(1) NOT NULL,
  CONTA_AUTOMATICA_EMP VARCHAR(1) NOT NULL,
  ELIMINADO_EMP VARCHAR(1) NOT NULL,
  USUARIO_EMP VARCHAR(25) NOT NULL,
  FECHA_EM DATE NOT NULL,
  IP_EMP VARCHAR(50) NOT NULL,
  PRIMARY KEY (ID_EMP)
);


CREATE TABLE CGB.SEG_EMPRESAS_L (
  ID_EMP INT(11) UNSIGNED NOT NULL,
  NOMBRE_EMP VARCHAR(30) NOT NULL,
  PAIS_EMP VARCHAR(3) NOT NULL,
  EMPRESA_EMP VARCHAR(3) NOT NULL,
  DEPARTAMENTO_EMP INT(11) UNSIGNED NOT NULL,
  PERIODO_EMP VARCHAR(5) NOT NULL,
  TIPO_CONTA_EMP VARCHAR(1) NOT NULL,
  CONTA_AUTOMATICA_EMP VARCHAR(1) NOT NULL,
  ELIMINADO_EMP VARCHAR(1) NOT NULL,
  USUARIO_EMP VARCHAR(25) NOT NULL,
  FECHA_EM DATE NOT NULL,
  IP_EMP VARCHAR(50) NOT NULL,
  FECHA_LOG_EMP TIMESTAMP NOT NULL,
  UDI_EMP VARCHAR(1) NOT NULL,
  PRIMARY KEY (ID_EMP, FECHA_LOG_EMP)
);


CREATE TABLE CGB.SEG_USUARIOS (
  ID_USU INT(11) UNSIGNED NOT NULL AUTO_INCREMENT,
  ID_EMP_USU INT(11) UNSIGNED NOT NULL,
  NOMBRE_USU VARCHAR(60) NOT NULL,
  CODIGO_USU VARCHAR(20) NOT NULL,
  PUESTO_USU VARCHAR(50) NOT NULL,
  CORREO_USU VARCHAR(60) NOT NULL,
  SUCURSAL_USU VARCHAR(50) NOT NULL,
  UNIDAD_NEGOCIO_USU VARCHAR(50) NOT NULL,
  SUPERVISOR_USU INT(11) UNSIGNED NOT NULL,
  ACTIVO_USU VARCHAR(1) NOT NULL,
  ELIMINADO_USU VARCHAR(1) NOT NULL,
  INSPECTOR_USU VARCHAR(1) NOT NULL,
  NOTIFICACION_USU VARCHAR(1) NOT NULL,
  CLAVE_USU VARCHAR(100) NOT NULL,
  USUARIO_USU VARCHAR(25) NOT NULL,
  FECHA_USU DATE NOT NULL,
  IP_USU VARCHAR(50) NOT NULL,
  PRIMARY KEY (ID_USU)
);


CREATE TABLE CGB.SEG_USUARIOS_L (
  ID_USU INT(11) UNSIGNED NOT NULL,
  NOMBRE_USU VARCHAR(60) NOT NULL,
  CODIGO_USU VARCHAR(20) NOT NULL,
  PUESTO_USU VARCHAR(50) NOT NULL,
  CORREO_USU VARCHAR(60) NOT NULL,
  SUCURSAL_USU VARCHAR(50) NOT NULL,
  UNIDAD_NEGOCIO_USU VARCHAR(50) NOT NULL,
  SUPERVISOR_USU INT(11) UNSIGNED NOT NULL,
  ACTIVO_USU VARCHAR(1) NOT NULL,
  ELIMINADO_USU VARCHAR(1) NOT NULL,
  FECHA_CREACION DATETIME NOT NULL,
  FECHA_MODIFICACION DATETIME NOT NULL,
  PRIMARY KEY (ID_USU),
  FOREIGN KEY (SUPERVISOR_USU) REFERENCES SEG_USUARIOS_L(ID_USU)
  );



TABLE CGB.GEN_MATERIALES (

TABLE CGB.GEN_MATERIALES_L (

  DESCRIPCION_MAT VARCHAR(50) NOT NULL,
  ELIMINADO_MAT VARCHAR(1) NOT NULL,
  USUARIO_MAT VARCHAR(25) NOT NULL,
